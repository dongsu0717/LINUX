SAMBA   
yum -y install samba   
smbpasswd -a root  
vi /etc/samba/smb.conf  
```sh
[root]
        comment = "코멘트"
        path = /root  # 경로
        valid users = root <user> # 접근 가능한 유저
        read only = no  # 읽기 전용
        writable = yes  # 쓰기 가능
        public = no
        browseable =yes
        printable = no
        create mask = 0750
```
firewall-cmd --permanent --zone=public --add-service=samba  
firewall-cmd --reload  
systemctl enable smb	systemctl start smb
